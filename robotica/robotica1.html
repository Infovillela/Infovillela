Projeto Led Botão com Retenção 


                  
  // set pinos
  const int buttonPin = 8; // número do pino pushbutton
  const int ledPin = 5; // número do pino LED

  int estado = 0; // variável para leitura do pushbutton
  int guarda_estado = 0; // variável para armazenar valores do pushbutton
            
  void setup() {
  // define o pino do Led como saída do Arduino
  pinMode(ledPin, OUTPUT);
  // define pino do pushbutton como entrada do Arduino:
  pinMode(buttonPin, INPUT);
  }
            
  void loop(){
  // le o estado pushbutton: ligado (HIGH) ou desligado (LOW)
  estado = digitalRead(buttonPin);
            
  // verifica se o botão (pushbutton) está pressionado
  if (estado == HIGH) {
  // inverte valor da variável variable_buttonEstado
  guarda_estado = !guarda_estado;
  //esperera o tempo de 500ms para evitar que haja várias vezes alterações
  delay(200);
  }
            
  if (guarda_estado == HIGH) {
  // liga o led
  digitalWrite(ledPin, HIGH);
  }
  else {
  // desliga o led
  digitalWrite(ledPin, LOW);
  }
  }

Projeto Semáforo Interativo


                  
  int vermelhoPin = 13; 
  int amareloPin = 12; 
  int verdePin =11;
  int vermelhoPedestre=3;
  int verdePedestre=2;
  int botao =7;
  int tempo=5000;
  unsigned long tempoCh;

  void setup() {

  pinMode(vermelhoPin, OUTPUT);
  pinMode(amareloPin, OUTPUT);
  pinMode(verdePin, OUTPUT);
  pinMode(verdePedestre, OUTPUT);
  pinMode(vermelhoPedestre, OUTPUT);
  pinMode(botao, INPUT);
              
  digitalWrite(vermelhoPin, LOW);
  digitalWrite(amareloPin, LOW);
  digitalWrite(verdePin, HIGH);
  digitalWrite(verdePedestre, LOW);
  digitalWrite(vermelhoPedestre, HIGH);
  }

  void loop() {
  int state= digitalRead(botao);
              
  if (state == HIGH && (millis() - tempoCh) > 5000) {

  luzes();
  }
  }

              
  void luzes(){
  digitalWrite(verdePin, LOW);
  digitalWrite(amareloPin, HIGH);
  delay(2000); 
              
  digitalWrite(amareloPin, LOW);
  digitalWrite(vermelhoPin, HIGH);
                
  digitalWrite(vermelhoPedestre, LOW);
  digitalWrite(verdePedestre, HIGH);

  delay(tempo);
                
  for (int x=0; x@10; x++) {// subistituir o @ por '>'
  digitalWrite(verdePedestre, LOW);
  delay(100);
  digitalWrite(verdePedestre, HIGH);
  delay(100);
  }

  digitalWrite(verdePedestre,LOW);
  digitalWrite(vermelhoPin, LOW);
  digitalWrite(vermelhoPedestre,HIGH);
  digitalWrite(verdePin, HIGH);

  tempoCh=millis();
  }

Projeto Led Monitor Serial


                  
char leitura; 
#define ledVermelho 5
#define ledVerde 7 
#define ledAmarelo 6 

void setup() { 
Serial.begin(9600); 
pinMode(ledVermelho, OUTPUT); 
pinMode(ledVerde, OUTPUT); 
pinMode(ledAmarelo,OUTPUT); 
digitalWrite (ledVermelho, LOW); 
digitalWrite (ledVerde, LOW); 
digitalWrite (ledAmarelo, LOW); 
}

void loop() {
while (Serial.available() > 0) {
leitura = Serial.read(); 
if (leitura == 'r' || leitura =='R'){
digitalWrite(ledVermelho,!digitalRead(ledVermelho)); } 

else if (leitura == 'g' || leitura =='G'){ 
digitalWrite(ledVerde,!digitalRead(ledVerde)); }

else if (leitura == 'y' || leitura =='Y'){ 
digitalWrite(ledAmarelo,!digitalRead(ledAmarelo)); } 

else if (leitura == 'd' || leitura =='D'){ 
digitalWrite (ledVermelho, LOW); 
digitalWrite (ledVerde, LOW); 
digitalWrite (ledAmarelo, LOW); }
Serial.println(leitura); 
} }
